import os
import javalang

def extract_all_methods_from_repo(path_to_repository):
    """
    Extracts all methods from the file content
    :param path_to_repository: string
    :return: list of strings
    """

    methods_in_repo = {}

    path_to_files = os.path.join(path_to_repository, "src", "main", "java")

    # walk through the files
    for root, _, files in os.walk(path_to_files):
        for java_file in files:
            if java_file.endswith(".java"):
                # read the file
                with open(os.path.join(root, java_file), "r", encoding="utf8") as f:
                    file_content = f.read()

                # extract all methods
                methods_in_file = extract_all_methods_from_file(file_content)

                if len(methods_in_file) > 0:
                    methods_in_repo[java_file] = methods_in_file

    return methods_in_repo


def extract_code_block(code: str, start_position: int) -> tuple[str, int] | None:
    """
    Extracts the code block starting at the given position according to the opening and closing braces
    :param code: Java or Kotlin code
    :param start_position: position of the opening brace of the code block
    :return: code block and end position of the code block
    """
    open_braces = 0
    code_block_start = start_position
    code_block_end = -1

    for i in range(start_position, len(code)):
        if code[i] == "{":
            open_braces += 1
        elif code[i] == "}":
            open_braces -= 1
            if open_braces == 0:
                code_block_end = i + 1
                break

    if code_block_end == -1:
        return None

    while code[code_block_start] == "\n":
        code_block_start += 1

    return code[code_block_start:code_block_end], code_block_end



def extract_all_methods_from_file(java_source: str) -> str | None:
    """
    This function extracts the code of the method with the
    given name and class from the Java source code
    :param java_source: Java source code
    :return: code of the method
    """
    tree = javalang.parse.parse(java_source)

    methods_in_file = []

    for _, node in tree.filter(javalang.tree.MethodDeclaration):
        start_line = node.position.line - 1
        start_position = sum(len(line) + 1 for line in java_source.splitlines()[:start_line]) \
                            + node.position.column - 1
        code_block = extract_code_block(java_source, start_position)
        if code_block is not None:
            methods_in_file.append(code_block[0])

    return methods_in_file

from datasets import load_dataset
from torch.utils.data import DataLoader
import torch
from transformers import AutoTokenizer, AutoModel
from tqdm.notebook import tqdm


def get_pretrained(name, output_size=768):
    _tokenizer = AutoTokenizer.from_pretrained(name)
    _model = AutoModel.from_pretrained(name)
    _batchnorm = torch.nn.BatchNorm1d(output_size, momentum=1, affine=False)
    
    batch_size = 4
    num_batches_to_iterate = 100 # number of batches to load and process

    java_methods_test_dataset = load_dataset("anjandash/java-8m-methods-v2", split="test", streaming=True)
    dataloader = DataLoader(java_methods_test_dataset, batch_size=batch_size)

    _batchnorm.train()
    
    all_embs = []
    for batch_idx, batch_data in tqdm(enumerate(dataloader), total=num_batches_to_iterate, desc='iterating over batches to init batch norm'):
        if batch_idx >= num_batches_to_iterate:
            break
        tokenized_batch = _tokenizer(batch_data['text'], padding=True, return_tensors="pt", truncation=True, max_length=512)
        embs = _model(**tokenized_batch).last_hidden_state.mean(dim=1)
        all_embs.append(embs)
    all_embs = torch.cat(all_embs)
    _batchnorm(all_embs)

    _batchnorm.eval()

    return _tokenizer, _model, _batchnorm

Firstly, I am trying the approach when the loss is calculated concerning each pair of methods. Cosine similarity belongs to segment [-1; +1], so we can use binary cross entropy loss after adding 1 and dividing by 2.

$$Loss = \frac{2}{N(N-1)}\sum_{i, j, i < j} -\log \left(\frac{cos\_sim(emb_i, emb_j) + 1}{2}\right)[file_i = file_j] - \log \left(1 - \frac{cos\_sim(emb_i, emb_j) + 1}{2}\right)[file_i \neq file_j] $$

import torch
from torch.nn.functional import cosine_similarity
from torch.nn.functional import binary_cross_entropy
from tqdm.notebook import tqdm

opt = torch.optim.Adam(model.parameters(), lr=1e-4)
epochs = 15

for epoch in tqdm(range(epochs)):

    embeddings = [
        batchnorm(model(**(X[i])).last_hidden_state.mean(dim=1)) for i in range(n_files)
    ]

    loss = 0

    for i in range(n_files):
        for j in range(i + 1, n_files):
            similarity = (cosine_similarity(embeddings[i], embeddings[j]) + 1) / 2 # normalize to [0, 1]
            # since similarity is in [0, 1], we can use binary cross entropy
            loss += binary_cross_entropy(similarity, torch.tensor([float(y[i] == y[j])]))

    loss /= n_files * (n_files - 1) / 2  # mean value

    opt.zero_grad()
    loss.backward()
    opt.step()

    print(f"Epoch: {epoch}, loss: {loss.item()}")

The first approach has a problem: files with a bigger number of methods dominate files with a smaller number of methods (we can see how the biggest 3-rd file dominated others).

To avoid this problem, I introduce a weight for each pair of methods. The weight is inversely proportional to the number of pairs of methods from the same files. Specifically, $$ \frac{1}{w_{i, j}} \propto \begin{cases} \frac{f_i (f_i - 1)}{2} & \text{methods i and j are from the same files} \\ f_i f_j & \text{if } \text{methods i and j are from different files} \end{cases} $$

where $f_i$ is a number of files in file of method i. Also, sum of weight is equal to 1 for fair comparison with the first approach (previously, all weight were equal and sumed up to 1). Then the loss will be

$$ \sum_{i, j, i < j} w_{i, j} (-\log \left(\frac{cos\_sim(emb_i, emb_j) + 1}{2}\right)[file_i = file_j] - \log \left(1 - \frac{cos\_sim(emb_i, emb_j) + 1}{2}\right)[file_i \neq file_j]) $$

count_same_file = [(y == i).sum() for i in y]

# weight for each pair in the loss
weight = np.zeros((n_files, n_files))
for i in range(n_files):
    for j in range(i + 1, n_files):
        if y[i] == y[j]:
            weight[i, j] = 1 / ((count_same_file[i] * (count_same_file[i] - 1)) / 2)
        else:
            weight[i, j] = 1 / (count_same_file[i] * count_same_file[j])

# sum of all weights should still be 1 for fair comparison
weight /= weight.sum()
